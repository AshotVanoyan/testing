#include <iostream>

double inc_dec(double p){
    return 100*p/(100+p);
}

double dec_inc(double p){
    return 100*p/(100-p);
}

double increase(double money, double percent){
    return money+money*percent/100;
}

double decrease(double money, double percent){
    return money-money*percent/100;
}

int main()
{
    double initialMoney=7000;
    double increaseFirst=25;
    double decreaseSecond=inc_dec(increaseFirst);
    
    std::cout<<decrease(increase(initialMoney, increaseFirst),decreaseSecond)<<std::endl;
    
    double decreaseFirst=34;
    double increaseSecond=dec_inc(decreaseFirst);
    
    std::cout<<increase(decrease(initialMoney,decreaseFirst),increaseSecond)<<std::endl;
   
    return 0;
}
